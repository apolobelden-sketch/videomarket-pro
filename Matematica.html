// ===== MATH MODULE ===== let mathScore = 0; let mathStreak = 0; let currentFraction = {}; let currentQuickMath = {}; function initMath() { newFraction(); newQuickMath(); } // ===== FRACTIONS GAME ===== function newFraction() { // Generar fracciones aleatorias con denominadores comunes (2, 3, 4, 5, 6, 8, 10) const dens = [2, 3, 4, 5, 6, 8, 10]; const den1 = dens[Math.floor(Math.random() * dens.length)]; const den2 = dens[Math.floor(Math.random() * dens.length)]; // Numeradores menores que denominadores const num1 = Math.floor(Math.random() * (den1 - 1)) + 1; const num2 = Math.floor(Math.random() * (den2 - 1)) + 1; currentFraction = { num1, den1, num2, den2 }; // Mostrar document.getElementById('fracNum1').textContent = num1; document.getElementById('fracDen1').textContent = den1; document.getElementById('fracNum2').textContent = num2; document.getElementById('fracDen2').textContent = den2; // Limpiar inputs document.getElementById('mathAnswerNum').value = ''; document.getElementById('mathAnswerDen').value = ''; } function checkFraction() { const num = parseInt(document.getElementById('mathAnswerNum').value); const den = parseInt(document.getElementById('mathAnswerDen').value); if (!num || !den || den === 0) { showToast('⚠️ Ingresa numerador y denominador'); return; } // Calcular resultado correcto: a/b + c/d = (ad + bc) / bd const correctNum = (currentFraction.num1 * currentFraction.den2) + (currentFraction.num2 * currentFraction.den1); const correctDen = currentFraction.den1 * currentFraction.den2; // Simplificar ambos para comparar const userSimplified = simplifyFraction(num, den); const correctSimplified = simplifyFraction(correctNum, correctDen); if (userSimplified.num === correctSimplified.num && userSimplified.den === correctSimplified.den) { mathScore += 10; mathStreak++; document.getElementById('mathScore').textContent = mathScore; document.getElementById('mathStreak').textContent = mathStreak; showToast('✅ ¡Correcto! +10 puntos'); if (mathStreak % 5 === 0) { showToast('🔥 ¡Racha de ' + mathStreak + '! ¡Sigue así!'); } setTimeout(newFraction, 1500); } else { mathStreak = 0; document.getElementById('mathStreak').textContent = mathStreak; showToast(`❌ Era ${correctSimplified.num}/${correctSimplified.den}`); } } function simplifyFraction(num, den) { const gcd = (a, b) => b === 0 ? a : gcd(b, a % b); const divisor = gcd(Math.abs(num), Math.abs(den)); return { num: num / divisor, den: den / divisor }; } // ===== QUICK MATH GAME ===== function newQuickMath() { const operations = ['+', '-', '×']; const op = operations[Math.floor(Math.random() * operations.length)]; let num1, num2, answer; switch(op) { case '+': num1 = Math.floor(Math.random() * 20) + 1; num2 = Math.floor(Math.random() * 20) + 1; answer = num1 + num2; break; case '-': num1 = Math.floor(Math.random() * 20) + 10; num2 = Math.floor(Math.random() * num1); answer = num1 - num2; break; case '×': num1 = Math.floor(Math.random() * 10) + 1; num2 = Math.floor(Math.random() * 10) + 1; answer = num1 * num2; break; } currentQuickMath = { num1, num2, op, answer }; document.getElementById('quickProblem').textContent = `${num1} ${op} ${num2} = ?`; // Generar opciones const options = [answer]; while (options.length < 4) { const fake = answer + Math.floor(Math.random() * 20) - 10; if (fake !== answer && fake >= 0 && !options.includes(fake)) { options.push(fake); } } // Mezclar opciones options.sort(() => Math.random() - 0.5); // Renderizar botones const container = document.getElementById('quickOptions'); container.innerHTML = options.map(opt => ` `).join(''); } function checkQuickMath(selected) { if (selected === currentQuickMath.answer) { mathScore += 5; mathStreak++; document.getElementById('mathScore').textContent = mathScore; document.getElementById('mathStreak').textContent = mathStreak; showToast('✅ ¡Rápido! +5 puntos'); setTimeout(newQuickMath, 1000); } else { mathStreak = 0; document.getElementById('mathStreak').textContent = mathStreak; showToast('❌ ¡Inténtalo de nuevo!'); } } // ===== MATH UTILITIES ===== function resetMath() { mathScore = 0; mathStreak = 0; document.getElementById('mathScore').textContent = '0'; document.getElementById('mathStreak').textContent = '0'; newFraction(); newQuickMath(); showToast('🔄 Juego reiniciado'); }
